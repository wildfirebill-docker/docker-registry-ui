Getting Started

Docker Run

# Create config
cat > registries.config.json <<'JSON'
{
  "registries": [
    {
      "name": "Local Registry",
      "api": "http://localhost:5001"
    }
  ]
}
JSON

# Start test registry
docker run -d --name test-registry -p 5001:5000 \
  -e REGISTRY_STORAGE_DELETE_ENABLED=true registry:2

# Run Registry UI (use environment variable)
docker run -d --name registry-ui -p 5000:5000 \
  -e 'REGISTRIES=[{"name":"Local Registry","api":"http://localhost:5001"}]' \
  -v $(pwd)/data:/app/data \
  ghcr.io/vibhuvioio/docker-registry-ui:latest
